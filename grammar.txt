file = [header CLRF] record *(CLRF record) [CLRF]

header = name *(COMMA name)

record = field *(COMMA field) / enclosed_field *(COMMA enclosed_field)

name = field

enclosed_field = DQUOTE *(TEXTDATA / COMMA) DQUOTE 

field = *(TEXTDATA) 

COMMA = %x2C

CR = %x0D

DQUOTE = %x22

LF = %x0A 
   
CRLF = CR LF 

TEXTDATA =  %x20-21 / %x23-2B / %x2D-7E
